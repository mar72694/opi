import javafx.application.Application; import javafx.scene.Group; import javafx.scene.PerspectiveCamera; import javafx.scene.Scene; import javafx.scene.paint.Color; import javafx.scene.shape.Box; import javafx.scene.transform.Rotate; import javafx.stage.Stage; public class House3D extends Application { @Override public void start(Stage primaryStage) { Group root = new Group(); // 房子主体 Box house = new Box(5, 5, 5); house.setTranslateX(0); house.setTranslateY(0); house.setTranslateZ(0); house.setMaterial(new javafx.scene.paint.PhongMaterial(Color.SADDLEBROWN)); root.getChildren().add(house); // 屋顶 Box roof = new Box(7, 1, 7); roof.setTranslateX(0); roof.setTranslateY(5); roof.setTranslateZ(0); roof.setRotateX(-30); roof.setMaterial(new javafx.scene.paint.PhongMaterial(Color.RED)); root.getChildren().add(roof); // 门 Box door = new Box(1, 3, 1); door.setTranslateX(2); door.setTranslateY(0); door.setTranslateZ(-2.5); door.setMaterial(new javafx.scene.paint.PhongMaterial(Color.DARKGRAY)); root.getChildren().add(door); // 窗户 1 Box window1 = new Box(1, 1, 1); window1.setTranslateX(-1.5); window1.setTranslateY(2); window1.setTranslateZ(-2.5); window1.setMaterial(new javafx.scene.paint.PhongMaterial(Color.LIGHTBLUE)); root.getChildren().add(window1); // 窗户 2 Box window2 = new Box(1, 1, 1); window2.setTranslateX(1.5); window2.setTranslateY(2); window2.setTranslateZ(-2.5); window2.setMaterial(new javafx.scene.paint.PhongMaterial(Color.LIGHTBLUE)); root.getChildren().add(window2); PerspectiveCamera camera = new PerspectiveCamera(true); camera.setTranslateZ(-10); Scene scene = new Scene(root, 800, 600, true); scene.setCamera(camera); primaryStage.setTitle("3D House"); primaryStage.setScene(scene); primaryStage.show(); } public static void main(String[] args) { launch(args);
